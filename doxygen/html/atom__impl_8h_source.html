32 inline static void parseMap(const string& options, _out_ map<string, string>& res) {
33  if (options.empty())
34  return;
35 
36  strtokenizer tokens(options, ",");
37  const int c = tokens.count_tokens();
38  for (int i = 0; i < c; i++) {
39  string tok = tokens.next_token();
40  trim(tok);
41  vector<string> optarg;
42  split(tok, '=', optarg);
43  if (!optarg.empty()) {
44  if (optarg.size() == 2)
45  res[trim(optarg[0])] = trim(optarg[1]);
46  else if (optarg.size() == 1)
47  res[trim(optarg[0])] = "";
48  }
49  }
50 }
