258 
259  if (i < len) {

262  return string({_str[_pos++]});
263  for (_pos++; _pos < len; _pos++)
264  if (_del.find(_str[_pos]) != string::npos)
265  return _str.substr(i, _pos - i);
266  return _str.substr(i);
267  }
268 
269  while (i < len && _del.find(_str[i]) != string::npos)
270  i++;
271

273  if (i < len) {
274  for (_pos++; _pos < len; _pos++)
275  if (_del.find(_str[_pos]) != string::npos)

287 inline string& wide2utf8(const wchar_t* in, _out_ string& out) {
288  unsigned int codepoint = 0;
289  for (; *in != 0; ++in) {
290  if (*in >= 0xd800 && *in <= 0xdbff)
291  codepoint = ((*in - 0xd800) << 10) + 0x10000;
292  else {
293  if (*in >= 0xdc00 && *in <= 0xdfff)
294  codepoint |= *in - 0xdc00;
295  else
296  codepoint = *in;
297 
298  if (codepoint <= 0x7f)
299  out.append(1, static_cast<char>(codepoint));
300  else if (codepoint <= 0x7ff) {
301  out.append(1, static_cast<char>(0xc0 | ((codepoint >> 6) & 0x1f)));
302  out.append(1, static_cast<char>(0x80 | (codepoint & 0x3f)));
303  } else if (codepoint <= 0xffff) {
304  out.append(1, static_cast<char>(0xe0 | ((codepoint >> 12) & 0x0f)));
305  out.append(1, static_cast<char>(0x80 | ((codepoint >> 6) & 0x3f)));
306  out.append(1, static_cast<char>(0x80 | (codepoint & 0x3f)));
307  } else {
308  out.append(1, static_cast<char>(0xf0 | ((codepoint >> 18) & 0x07)));
309  out.append(1, static_cast<char>(0x80 | ((codepoint >> 12) & 0x3f)));
310  out.append(1, static_cast<char>(0x80 | ((codepoint >> 6) & 0x3f)));
311  out.append(1, static_cast<char>(0x80 | (codepoint & 0x3f)));
312  }
313  codepoint = 0;
314  }
315  }
316  return out;
317 }

328 inline wstring& utf82wide(const char* in, _out_ wstring& out) {
329  if (in == nullptr)
330  return out;
331  unsigned int codepoint;
332  while (*in != 0) {
333  unsigned char ch = static_cast<unsigned char>(*in);
334  if (ch <= 0x7f)
335  codepoint = ch;
336  else if (ch <= 0xbf)
337  codepoint = (codepoint << 6) | (ch & 0x3f);
338  else if (ch <= 0xdf)
339  codepoint = ch & 0x1f;
340  else if (ch <= 0xef)
341  codepoint = ch & 0x0f;
342  else
343  codepoint = ch & 0x07;
344  ++in;
345  if (((*in & 0xc0) != 0x80) && (codepoint <= 0x10ffff)) {
346  if (codepoint > 0xffff) {
347  out.append(1, static_cast<wchar_t>(0xd800 + (codepoint >> 10)));
348  out.append(1, static_cast<wchar_t>(0xdc00 + (codepoint & 0x03ff)));
349  } else if (codepoint < 0xd800 || codepoint >= 0xe000)
350  out.append(1, static_cast<wchar_t>(codepoint));
351  }
352  }
353  return out;
354 }
